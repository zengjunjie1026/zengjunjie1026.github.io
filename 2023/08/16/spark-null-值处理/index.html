def parseCheck(s: String,min:Double,max:Double): Option[Double] = {
      try {
        val va = s.toDouble
        if (va < min || va > max) {
          Some(va)
        }else {
          Some(Double.NaN)
        }
      }
      catch { case _ => Some(Double.NaN)
      }
    }
